#代码

```javascript

<!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd">
<html xmlns="http://www.w3.org/1999/xhtml">
<head>
<meta http-equiv="Content-Type" content="text/html; charset=utf-8" />
<title>简单的写字板</title>

    <style type="text/css">
        div{width:4px;height: 4px;background-color: red;position: absolute; border-radius: 2px;}
    </style>
</head>

<body>
<span>左键写，中间键删除</span><input type="button" onclick="flashBody()" value="清除"/>
</body>
<script type="text/javascript">
    /*
     * write by waitingbar
     * time 20121002
     *
     */

    var x , y;

    //当需求为获得的坐标值相对于body时，用：
    function positionBody(event){
        event = event||window.event;
        //获得相对于body定位的横标值；
        x=event.clientX
        //获得相对于body定位的纵标值；
        y=event.clientY


    }


    var flag = 0;

    //应用：
    document.onmousemove = function(event){
        positionBody(event);

        if(flag == 1){
            document.body.style = 'cursor:pointer';
            var box = document.createElement('div');
            box.style.top = y + 'px';
            box.style.left = x + 'px';
            document.body.appendChild(box);
        }else if(flag === 2){

            var _target = event.target;
            if(_target.style.top.length > 0){
                document.body.removeChild(_target)
            }
        }

    }

    document.onmousedown = function (event) {
        var event = event||window.event;
        var btnNum= event.button;
        if(btnNum == 0){
            flag = 1;
        }else if(btnNum == 1){
            flag = 2;

        }

    }

    document.onmouseup = function () {
        flag = 0;
    }

    function flashBody() {
        location.reload()
    }

</script>
</html>


    
```
